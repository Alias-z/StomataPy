
######### stomatapy (MacOSX) #########

1. download mambaforge
https://github.com/conda-forge/miniforge#mambaforge

2. instllation
bash Mambaforge-MacOSX-arm64.sh

3. add mamba to env
go to folder ~/.zshrc add 
export PATH="$PATH:/Users/zhang/mambaforge/bin"
export PATH="$PATH:/Users/Zhang/mambaforge/envs/mm/bin"

4. install pakages
zsh
mamba create -n mm python=3.8 -y
mamba activate mm
mamba install pytorch torchvision openmim tensorboard opencv jupyter more-itertools matplotlib openpyxl tqdm pandas scikit-image 
mamba install albumentations napari

mamba install imgviz mahotas numpy pillow pyqt pyyaml pycocotools -y
mamba install matplotlib onnxruntime onnx timm -y
